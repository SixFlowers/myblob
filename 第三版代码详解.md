### 设计对比：AnyBlob vs 本版本

#### AnyBlob 的设计

AnyBlob 的 Provider 基类**没有**单独的 `_https` 字段：

```cpp
class Provider {
protected:
    CloudService _type;  // 只有一个类型字段
    
public:
    CloudService getType() { return _type; }
};
```

判断 HTTPS 的方式：

```cpp
// 直接用枚举值判断
bool isHTTPS = (provider->getType() == Provider::CloudService::HTTPS);
```

#### 本版本的设计

为了对标 AnyBlob，本版本**也移除了** `_https` 字段：

```cpp
class Provider {
protected:
    CloudService type_;  // 只有一个类型字段
    
public:
    CloudService getType() const { return type_; }
    
    // ✅ 添加了 isHTTPS 方法，替代原来的 https_ 字段
    bool isHTTPS() const { return type_ == CloudService::HTTPS; }
};
```

**为什么这样设计？**

1. **简化状态**：避免维护两个相关的状态（`type_` 和 `https_`）
2. **单一数据源**：只有一个真实来源（`type_`），其他都是派生
3. **对标 AnyBlob**：保持与 AnyBlob 一样的设计思路
4. **避免不一致**：如果修改了 `type_` 但忘记更新 `https_`，会产生 bug

### Provider 实现文件（改进）

> 📁 文件：`src/cloud/provider.cpp`（**需要修改使用 ConnectionManager**）

```cpp
// ============================================================================
// provider.cpp - 云服务提供商实现（连接池版）
// ============================================================================

#include "cloud/provider.hpp"
#include "network/http_client.hpp"
#include "utils/data_vector.hpp"
#include <iostream>

namespace myblob::cloud {

// ============================================================================
// 构造函数
// ============================================================================

Provider::Provider(network::ConnectionManager& conn_mgr,
                   network::HttpClient& http_client,
                   CloudService type)
    : conn_mgr_(conn_mgr)
    , http_client_(http_client)
    , type_(type) {
    std::cerr << "[DEBUG] Provider: Created with ConnectionManager, type=" 
              << static_cast<int>(type_) << std::endl;
}

// ============================================================================
// 析构函数
// ============================================================================

Provider::~Provider() {
    std::cerr << "[DEBUG] Provider: Destroyed" << std::endl;
}

// ============================================================================
// download - 下载文件
// ============================================================================

network::HttpResponse Provider::download(const std::string& url, uint64_t offset, uint64_t length) {
    std::cerr << "[DEBUG] Provider::download: url=" << url << std::endl;
    
    // 判断是否使用 HTTPS
    bool use_tls = url.starts_with("https://");
    
    // ✅ 正确用法：从连接池获取连接
    auto connection = conn_mgr_.getConnection(address_, port_, use_tls);
    
    if (!connection) {
        std::cerr << "[ERROR] Failed to get connection from pool" << std::endl;
        return network::HttpResponse{};
    }
    
    // ✅ 使用连接发送请求（通过 HttpClient）
    network::HttpResponse response = http_client_.send(
        connection,
        "GET",
        "/",
        offset,
        length
    );
    
    // ✅ 归还连接到连接池
    conn_mgr_.returnConnection(connection);
    
    if (!response.isSuccess()) {
        std::cerr << "[ERROR] Download failed: status=" << response.getStatusCode() << std::endl;
    }
    
    return response;
}

// ============================================================================
// HTTPProvider 构造函数
// ============================================================================

HTTPProvider::HTTPProvider(network::ConnectionManager& conn_mgr,
                             network::HttpClient& http_client,
                             const std::string& addr, 
                             uint16_t port, 
                             bool https)
    : Provider(conn_mgr, http_client, https ? CloudService::HTTPS : CloudService::HTTP)
    , address_(addr)
    , port_(port) {
    std::cerr << "[DEBUG] HTTPProvider: Created for " << addr << ":" << port << std::endl;
}

// ============================================================================
// createGetRequest - 创建 GET 请求
// ============================================================================

std::unique_ptr<utils::DataVector<uint8_t>> HTTPProvider::createGetRequest(
    const std::string& file_path,
    uint64_t offset,
    uint64_t length
) {
    // 构建 HTTP 请求
    network::HttpRequest request;
    request.method = network::HttpRequest::Method::GET;
    request.path = file_path;
    
    // 添加 Range 头
    if (offset > 0 || length > 0) {
        std::string range = "bytes=" + std::to_string(offset) + "-";
        if (length > 0) {
            range += std::to_string(offset + length - 1);
        }
        request.headers["Range"] = range;
    }
    
    // 构建请求字符串
    std::string request_str = "GET " + file_path + " HTTP/1.1\r\n";
    for (const auto& header : request.headers) {
        request_str += header.first + ": " + header.second + "\r\n";
    }
    request_str += "\r\n";
    
    auto result = std::make_unique<utils::DataVector<uint8_t>>();
    result->resize(request_str.size());
    std::memcpy(result->data(), request_str.data(), request_str.size());
    
    return result;
}

}  // namespace myblob::cloud
```

```cpp
// ============================================================================
// example.cpp - 示例程序（连接池版）
// ============================================================================

#include "cloud/provider.hpp"
#include "network/connection_manager.hpp"
#include "network/http_client.hpp"
#include <iostream>
#include <fstream>

int main() {
    std::cout << "=== AnyBlob Connection Pool Demo ===" << std::endl;
    
    // ✅ 步骤 1：创建连接池
    myblob::network::ConnectionManager conn_mgr(100, 300, 10);
    
    // ✅ 步骤 2：创建 HTTP 客户端（传入连接池）
    myblob::network::HttpClient http_client(conn_mgr);
    
    // ✅ 步骤 3：创建 Provider（使用连接池和 HTTP 客户端）
    myblob::cloud::HTTPProvider provider(
        conn_mgr,                // 连接池管理器
        http_client,              // HTTP 客户端
        "example.com",           // 地址
        443,                     // 端口（HTTPS 用 443）
        true                     // 使用 HTTPS
    );
    
    std::string output = "output.html";
    
    std::cout << "Downloading: https://example.com/" << std::endl;
    
    // ✅ 步骤 4：下载文件（使用连接池）
    auto response = provider.download("https://example.com/");
    
    if (response.isSuccess()) {
        std::cout << "Download successful! Status: " << response.getStatusCode() << std::endl;
        
        // 保存到文件
        std::ofstream outfile(output, std::ios::binary);
        if (outfile && response.body.size() > 0) {
            outfile.write(reinterpret_cast<const char*>(response.body.data()), 
                         response.body.size());
            std::cout << "Saved to: " << output << std::endl;
        }
        
        // 打印连接池统计
        auto stats = conn_mgr.getStats();
        std::cout << "Connection pool stats:" << std::endl;
        std::cout << "  Total: " << stats.total_connections << std::endl;
        std::cout << "  Idle: " << stats.idle_connections << std::endl;
        std::cout << "  In use: " << stats.in_use_connections << std::endl;
    } else {
        std::cout << "Download failed! Status: " << response.getStatusCode() << std::endl;
        return 1;
    }
    
    return 0;
}
```

## 第三版概述

### 什么是第三版？

第三版是第一版的**连接池版**，解决性能问题：

| 问题 | 第二版 | 第三版 | 影响 |
|------|--------|--------|------|
| **连接复用** | 每次创建新连接 | 连接池复用 | 性能提升 10-100 倍 |
| **线程安全** | 不支持多线程 | 线程安全 | 可并发下载 |
| **连接管理** | 分散管理 | 集中管理 | 资源可控 |
| **连接复用** | 无法复用 | 复用已有连接 | 减少握手次数 |

### 学习目标

完成第三版后，你应该理解：

1. **连接池**：什么是连接池，为什么需要连接池
2. **线程安全**：如何使用 mutex 保护共享资源
3. **连接状态**：连接的生命周期管理
4. **资源管理**：如何控制连接数量

## 核心概念详解

### 什么是连接池？

#### 没有连接池时

```
┌─────────────────────────────────────────────────────────┐
│  请求 1: client → [创建连接] → 下载 → [关闭连接]         │
│  请求 2: client → [创建连接] → 下载 → [关闭连接]         │
│  请求 3: client → [创建连接] → 下载 → [关闭连接]         │
└─────────────────────────────────────────────────────────┘

❌ 问题：
   - 每次都要 DNS 解析
   - 每次都要 TCP 三次握手
   - 每次都要 TLS 握手（HTTPS）
   - 浪费时间！
```

#### 有连接池时

```
┌─────────────────────────────────────────────────────────┐
│                                                          │
│    ┌─────────────────────────────────┐                   │
│    │      Connection Pool (连接池)    │                   │
│    │  ┌─────┐ ┌─────┐ ┌─────┐       │                   │
│    │  │Conn1│ │Conn2│ │Conn3│ ...  │                   │
│    │  └─────┘ └─────┘ └─────┘       │                   │
│    └─────────────────────────────────┘                   │
│           ↑              ↓              ↓                │
│  请求 1 ──┘  ──→ 下载 ──→ 归还连接                      │
│  请求 2 ──┴─→ 下载 ──→ 归还连接                          │
│  请求 3 ──────────→ 下载 ──→ 归还连接                    │
│                                                          │
└─────────────────────────────────────────────────────────┘

✅ 优点：
   - 连接复用，不用每次创建/关闭
   - 减少 DNS、TCP、TLS 握手次数
   - 性能提升 10-100 倍
```

### 连接的生命周期

```
创建 ──→ 连接中 ──→ 空闲 ──→ 使用中 ──→ 空闲 ──→ ... ──→ 关闭
  │           │         │        │        │              │
  │           │         │        │        │              │
  ▼           ▼         ▼        ▼        ▼              ▼
socket()   connect()  等待使用  正在下载  归还到池      close()
```

| 状态 | 说明 |
|------|------|
| `IDLE` | 空闲可用，可以被获取使用 |
| `IN_USE` | 正在被某个请求使用 |
| `CONNECTING` | 正在建立连接（DNS/TCP/TLS） |
| `CLOSED` | 已关闭，不能再使用 |

### 线程安全问题

```
                    ┌──────────────┐
                    │  Thread 1    │
                    │  getConn()   │
                    │      ↓       │
         ┌──────────┼──────────────┼──────────┐
         │          │              │          │
         ▼          │              │          ▼
    ┌─────────┐     │              │     ┌─────────┐
    │ Conn 1  │◄────┘              └────►│ Conn 2  │
    │ IN_USE  │                        │  IDLE   │
    └─────────┘                        └─────────┘
         ▲                                      │
         │                                      │
         │         ┌──────────────┐             │
         │         │  Thread 2    │             │
         │         │ returnConn() │             │
         │         └──────────────┘             │
         │                                      │
         └──────────────────────────────────────┘
                          │
                          ▼
                   ┌─────────────┐
                   │   Mutex     │ ← 保护连接池！
                   │ 锁住后操作  │
                   └─────────────┘
```

**为什么需要锁？**
- 多个线程同时访问连接池
- 如果不加锁，可能两个线程拿到同一个连接
- 或者一个线程关闭连接，另一个线程还在使用

## 代码实现详解

### Connection 头文件

> 📁 文件：`include/network/connection.hpp`（**新建此文件**）

```cpp
// ============================================================================
// connection.hpp - 连接对象
// ============================================================================
// 功能：封装一个 TCP/HTTPS 连接
// 包含：socket fd、目标地址、连接状态、SSL 对象
// ============================================================================

#pragma once
#include <cstdint>
#include <string>
#include <atomic>
#include <openssl/ssl.h>

namespace myblob::network {

// ============================================================================
// ConnectionState - 连接状态枚举
// ============================================================================

enum class ConnectionState {
    IDLE,        // 空闲可用
    IN_USE,      // 正在使用
    CONNECTING,  // 正在连接
    CLOSED       // 已关闭
};

// ============================================================================
// Connection 类
// ============================================================================

class Connection {
public:
    // ============================================================
    // 构造函数
    // ============================================================
    // 参数：
    //   host - 目标主机名
    //   port - 目标端口
    //   use_tls - 是否使用 HTTPS
    
    Connection(const std::string& host, uint16_t port, bool use_tls);
    
    // ============================================================
    // 析构函数
    // ============================================================
```

```cpp
Connection::Connection(const std::string& host, uint16_t port, bool use_tls)
    : host_(host)

    , use_tls_(use_tls)
    , sockfd_(-1)
    , ssl_(nullptr)
    , ssl_context_(nullptr)
    , state_(ConnectionState::IDLE)
    , last_used_(time(nullptr))
    , connected_(false) {
    
    std::cerr << "[DEBUG] Connection: Creating for " << host_ << ":" << port_ 
              << " (TLS=" << use_tls_ << ")" << std::endl;
}

// ============================================================================
// 析构函数
// ============================================================================

Connection::~Connection() {
    disconnect();
}

// ============================================================================
// connect - 建立连接
// ============================================================================

bool Connection::connect() {
    std::cerr << "[DEBUG] Connection::connect: " << host_ << ":" << port_ << std::endl;
    
    // 已经在连接中
    if (state_.load() == ConnectionState::CONNECTING) {
        std::cerr << "[WARN] Connection already connecting" << std::endl;
        return false;
    }
    
    // 已经连接
    if (connected_ && sockfd_ >= 0) {
        std::cerr << "[DEBUG] Connection already established" << std::endl;
        return true;
    }
    
    // 标记为正在连接
    state_.store(ConnectionState::CONNECTING);
    
    // ============================================================
    // 第一步：创建 socket
    // ============================================================
    
    sockfd_ = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd_ < 0) {
        std::cerr << "[ERROR] socket() failed: " << strerror(errno) << std::endl;
        state_.store(ConnectionState::CLOSED);
        return false;
    }
    
    // 使用 Defer 确保清理
    Defer defer([this]() {
        if (sockfd_ >= 0 && !connected_) {
            close(sockfd_);
            sockfd_ = -1;
        }
    });
    
    // ============================================================
    // 第二步：设置 socket 选项
    // ============================================================
    
    // 设置超时（10 秒）
    struct timeval timeout;
    timeout.tv_sec = 10;
    timeout.tv_usec = 0;
    setsockopt(sockfd_, SOL_SOCKET, SO_SNDTIMEO, &timeout, sizeof(timeout));
    setsockopt(sockfd_, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout));
    
    // 禁用 Nagle 算法（低延迟）
    int flag = 1;
    setsockopt(sockfd_, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag));
    
    // ============================================================
    // 第三步：DNS 解析
    // ============================================================
    
    struct hostent* server = gethostbyname(host_.c_str());
    if (server == nullptr) {
        std::cerr << "[ERROR] DNS lookup failed for: " << host_ << std::endl;
        state_.store(ConnectionState::CLOSED);
        return false;
    }
    
    // ============================================================
    // 第四步：设置服务器地址
    // ============================================================
    
    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    memcpy(&addr.sin_addr.s_addr, server->h_addr, server->h_length);
    addr.sin_port = htons(port_);
    
    // ============================================================
    // 第五步：TCP 连接
    // ============================================================
    
    if (connect(sockfd_, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        std::cerr << "[ERROR] connect() failed: " << strerror(errno) << std::endl;
        state_.store(ConnectionState::CLOSED);
        return false;
    }
    
    std::cerr << "[DEBUG] TCP connected to " << host_ << ":" << port_ << std::endl;
    
    // ============================================================
    // 第六步：TLS 握手（HTTPS）
    // ============================================================
    
    if (use_tls_) {
        // 创建 SSL（需要调用者提供 SSL_CTX）
        ssl_ = SSL_new(static_cast<SSL_CTX*>(ssl_context_));
        if (!ssl_) {
            std::cerr << "[ERROR] SSL_new() failed" << std::endl;
            state_.store(ConnectionState::CLOSED);
            return false;
        }
        
        SSL_set_fd(ssl_, sockfd_);
        
        if (SSL_connect(ssl_) != 1) {
            std::cerr << "[ERROR] SSL handshake failed" << std::endl;
            SSL_free(ssl_);
            ssl_ = nullptr;
            state_.store(ConnectionState::CLOSED);
            return false;
        }
        
        std::cerr << "[DEBUG] TLS handshake successful" << std::endl;
    }
    
    // ============================================================
    // 连接成功！
    // ============================================================
    
    connected_ = true;
    state_.store(ConnectionState::IDLE);
    last_used_ = time(nullptr);
    
    // 提前 release，避免析构时关闭
    defer.release();
    
    std::cerr << "[DEBUG] Connection::connect: SUCCESS" << std::endl;
    return true;
}

// ============================================================================
// disconnect - 断开连接
// ============================================================================

void Connection::disconnect() {
    std::cerr << "[DEBUG] Connection::disconnect: " << host_ << ":" << port_ << std::endl;
    
    if (ssl_) {
        SSL_free(ssl_);
        ssl_ = nullptr;
    }
    
    if (sockfd_ >= 0) {
        close(sockfd_);
        sockfd_ = -1;
    }
    
    connected_ = false;
    state_.store(ConnectionState::CLOSED);
}
```

```cpp
// ============================================================================
// connection_manager.cpp - 连接池管理器实现
// ============================================================================

#include "network/connection_manager.hpp"
#include "utils/defer.hpp"
#include <openssl/ssl.h>
#include <algorithm>

namespace myblob::network {

using myblob::utils::Defer;

// ============================================================================
// 构造函数
// ============================================================================

ConnectionManager::ConnectionManager(
    size_t max_connections,
    int max_idle_seconds,
    int connect_timeout
) : max_connections_(max_connections)
  , max_idle_seconds_(max_idle_seconds)
  , connect_timeout_(connect_timeout)
  , ssl_context_(nullptr) {
    
    std::cerr << "[DEBUG] ConnectionManager: Initializing..." << std::endl;
    
    // 初始化 OpenSSL
    SSL_library_init();
    SSL_load_error_strings();
    OpenSSL_add_all_algorithms();
    
    // 创建 SSL 上下文
    ssl_context_ = SSL_CTX_new(TLS_client_method());
    if (!ssl_context_) {
        throw std::runtime_error("Failed to create SSL context");
    }
    
    std::cerr << "[DEBUG] ConnectionManager: Initialized (max_connections=" 
              << max_connections_ << ")" << std::endl;
}

// ============================================================================
// 析构函数
// ============================================================================

ConnectionManager::~ConnectionManager() {
    closeAll();
    
    if (ssl_context_) {
        SSL_CTX_free(static_cast<SSL_CTX*>(ssl_context_));
        ssl_context_ = nullptr;
    }
    
    std::cerr << "[DEBUG] ConnectionManager: Destroyed" << std::endl;
}

// ============================================================================
// getConnection - 获取连接（参考 Redis::GetConnection）
// ============================================================================

std::shared_ptr<Connection> ConnectionManager::getConnection(
    const std::string& host,
    uint16_t port,
    bool use_tls
) {
    std::cerr << "[DEBUG] ConnectionManager::getConnection: " << host << ":" << port 
              << " (TLS=" << use_tls << ")" << std::endl;
    
    std::unique_lock<std::mutex> lock(mutex_);
    
    // ============================================================
    // 等待空闲连接（参考 Redis 的条件变量）
    // ============================================================
    cond_.wait(lock, [this]() {
        if (stop_) return true;
        // 有空闲连接就返回
        for (auto& conn : pool_) {
            if (conn->getState() == ConnectionState::IDLE) {
                return true;
            }
        }
        // 未满也能创建
        return pool_.size() < max_connections_;
    });
    
    // 如果已停止，直接返回
    if (stop_) {
        std::cerr << "[WARN] ConnectionManager is stopped" << std::endl;
        return nullptr;
    }
    
    // ============================================================
    // 第一步：尝试复用现有连接
    // ============================================================
    
    auto conn = findExistingConnection(host, port, use_tls);
    if (conn) {
        std::cerr << "[DEBUG] Reusing existing connection" << std::endl;
        conn->markUsed();
        return conn;
    }
    
    // ============================================================
    // 第二步：创建新连接
    // ============================================================
    
    conn = createNewConnection(host, port, use_tls);
    if (!conn) {
        std::cerr << "[ERROR] Failed to create new connection" << std::endl;
        return nullptr;
    }
    
    pool_.push_back(conn);
    conn->markUsed();
    
    return conn;
}

// ============================================================================
// returnConnection - 归还连接（参考 Redis::ReturnConnection）
// ============================================================================

void ConnectionManager::returnConnection(std::shared_ptr<Connection> conn) {
    if (!conn) return;
    
    std::cerr << "[DEBUG] ConnectionManager::returnConnection: " << conn->getHost() 
              << ":" << conn->getPort() << std::endl;
    
    std::lock_guard<std::mutex> lock(mutex_);
    
    // 检查连接是否有效
    if (conn->getState() == ConnectionState::CLOSED) {
        auto it = std::find(pool_.begin(), pool_.end(), conn);
        if (it != pool_.end()) {
            pool_.erase(it);
            std::cerr << "[DEBUG] Removed closed connection from pool" << std::endl;
        }
        return;
    }
    
    // 标记为空闲
    conn->markIdle();
    
    // 通知等待的线程（参考 Redis）
    cond_.notify_one();
}

// ============================================================================
// closeIdleConnections - 关闭空闲连接
// ============================================================================

void ConnectionManager::closeIdleConnections() {
    std::lock_guard<std::mutex> lock(mutex_);
    
    std::cerr << "[DEBUG] ConnectionManager::closeIdleConnections: pool size before=" 
              << pool_.size() << std::endl;
    
    // 使用迭代器安全删除
    for (auto it = pool_.begin(); it != pool_.end(); ) {
        auto& conn = *it;
        
        // 检查是否空闲超时
        if (conn->isIdleTooLong(max_idle_seconds_)) {
            std::cerr << "[DEBUG] Closing idle connection to " << conn->getHost() << std::endl;
            conn->disconnect();
            it = pool_.erase(it);  // erase 返回下一个迭代器
        } else {
            ++it;
        }
    }
    
    std::cerr << "[DEBUG] ConnectionManager::closeIdleConnections: pool size after=" 
              << pool_.size() << std::endl;
}

// ============================================================================
// closeAll - 关闭所有连接（参考 Redis::Close）
// ============================================================================

void ConnectionManager::closeAll() {
    std::lock_guard<std::mutex> lock(mutex_);
    
    // 标记停止（参考 Redis 的 _b_stop）
    stop_ = true;
    
    // 唤醒所有等待的线程
    cond_.notify_all();
    
    std::cerr << "[DEBUG] ConnectionManager::closeAll: closing " << pool_.size()
              << " connections" << std::endl;
    
    for (auto& conn : pool_) {
        conn->disconnect();
    }
    
    pool_.clear();
}

// ============================================================================
// getStats - 获取统计信息
// ============================================================================

ConnectionManager::Stats ConnectionManager::getStats() const {
    std::lock_guard<std::mutex> lock(mutex_);
    
    Stats stats;
    stats.total_connections = pool_.size();
    stats.idle_connections = 0;
    stats.in_use_connections = 0;
    stats.max_connections = max_connections_;
    
    for (const auto& conn : pool_) {
        if (conn->getState() == ConnectionState::IDLE) {
            stats.idle_connections++;
        } else if (conn->getState() == ConnectionState::IN_USE) {
            stats.in_use_connections++;
        }
    }
    
    return stats;
}

// ============================================================================
// findExistingConnection - 查找现有连接
// ============================================================================

std::shared_ptr<Connection> ConnectionManager::findExistingConnection(
    const std::string& host,
    uint16_t port,
    bool use_tls
) {
    for (auto& conn : pool_) {
        // 检查是否匹配
        if (conn->getHost() == host && 
            conn->getPort() == port &&
            conn->getState() == ConnectionState::IDLE &&
            conn->usesTLS() == use_tls) {
            
            return conn;
        }
    }
    
    return nullptr;
}

// ============================================================================
// createNewConnection - 创建新连接
// ============================================================================

std::shared_ptr<Connection> ConnectionManager::createNewConnection(
    const std::string& host,
    uint16_t port,
    bool use_tls
) {
    auto conn = std::make_shared<Connection>(host, port, use_tls);
    
    // 设置 SSL 上下文
    conn->setSSLContext(ssl_context_);
    
    // 建立连接
    if (!conn->connect()) {
        return nullptr;
    }
    
    return conn;
}
```
