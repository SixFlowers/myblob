## 1. Socket 抽象基类

> 📁 文件：`include/network/socket.hpp`（**新建此文件**）

### 代码详解

```cpp
#pragma once
#include <chrono>      // 用于 std::chrono::milliseconds 超时设置
#include <cstdint>     // 用于 uint32_t 等类型

namespace myblob::network {

class Socket {
public:
    // ================================================================
    // 事件类型枚举：区分读事件和写事件
    // ================================================================
    enum class EventType : uint8_t {
        read = 0,   // 读事件：等待 socket 可读（收到数据）
        write = 1   // 写事件：等待 socket 可写（可以发送数据）
    };

    // ================================================================
    // 统一的请求结构
    // 使用 union 区分发送数据和接收数据，节省内存
    // ================================================================
    struct Request {
        union Data {
            const uint8_t* sendData;  // 发送时使用：指向要发送的数据
            uint8_t* recvData;        // 接收时使用：指向接收缓冲区
        };
        Data data;           // union 联合体
        int64_t length;      // 数据长度
        int32_t fd;          // socket 文件描述符
        EventType event;     // 事件类型：读或写
        void* userData;      // 用户数据指针，用于关联 OriginalMessage
    };

    // ================================================================
    // 虚析构函数：确保派生类正确析构
    // ================================================================
    virtual ~Socket() = default;

    // ================================================================
    // 异步发送/接收接口
    // msgFlags: POSIX send/recv 的标志位（如 MSG_DONTWAIT）
    // ================================================================
    virtual bool send(const Request& req, int32_t msgFlags = 0) = 0;
    virtual bool recv(Request& req, int32_t msgFlags = 0) = 0;

    // ================================================================
    // 带超时的发送/接收接口
    // timeout: 超时时间
    // ================================================================
    virtual bool send_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags = 0) = 0;
    virtual bool recv_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags = 0) = 0;

    // ================================================================
    // 获取完成的请求
    // 返回已完成的 Request 指针，nullptr 表示没有完成的请求
    // ================================================================
    virtual Request* complete() = 0;

    // ================================================================
    // 提交请求到内核
    // 返回已完成的请求数量，-1 表示错误
    // ================================================================
    virtual int32_t submit() = 0;
};

}  // namespace myblob::network
```

### 设计说明

| 组件 | 说明 |
|------|------|
| `EventType` | 区分读/写事件，用于 poll 监控 |
| `Request` | 统一请求结构，union 节省内存 |
| `userData` | 关联 OriginalMessage，用于回调 |
| 虚函数 | 确保派生类实现自己的多路复用逻辑 |

## 2. PollSocket - POSIX poll 多路复用

> 📁 文件：`include/network/poll_socket.hpp`（**新建此文件**）

### 代码详解

```cpp
#pragma once
#include "network/socket.hpp"          // 包含 Socket 基类
#include <chrono>                       // 超时时间点
#include <unordered_map>               // fd -> RequestInfo 映射
#include <vector>                     // pollfd 数组
#include <sys/poll.h>                 // POSIX poll 系统调用

namespace myblob::network {

class PollSocket : public Socket {
private:
    // ================================================================
    // 请求信息结构：存储请求的额外信息
    // ================================================================
    struct RequestInfo {
        Request* request;                                           // 关联的请求
        std::chrono::time_point<std::chrono::steady_clock> timeout;  // 超时时间点
        int32_t flags;                                              // 标志位
    };

    std::vector<Request*> ready;               // 已完成的请求队列
    std::unordered_map<int, RequestInfo> fdToRequest;  // fd 到请求的映射
    std::vector<pollfd> pollfds;                // poll 监控的 fd 数组
    int32_t readyFds = 0;                       // 已读取的 ready 索引
    int32_t submitted = 0;                      // 已提交请求计数

public:
    ~PollSocket() override = default;          // 默认析构函数

    // 发送请求：监控 socket 可写
    bool send(const Request& req, int32_t msgFlags = 0) override;
    // 接收请求：监控 socket 可读
    bool recv(Request& req, int32_t msgFlags = 0) override;
    // 带超时的版本
    bool send_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags = 0) override;
    bool recv_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags = 0) override;

    // 获取完成的请求
    Request* complete() override;
    // 提交并等待事件
    int32_t submit() override;

private:
    // 将 fd 加入 poll 监控
    void enqueue(int fd, short events, RequestInfo req);
};

}  // namespace myblob::network
```

## 3. PollSocket 实现

> 📁 文件：`src/network/poll_socket.cpp`（**新建此文件**）

### 代码详解

```cpp
#include "network/poll_socket.hpp"
#include <cerrno>           // 错误码
#include <cstring>          // strerror
#include <unistd.h>         // close

namespace myblob::network {

// ================================================================
// 将 fd 加入 poll 监控
// fd: socket 文件描述符
// events: POLLIN（可读）或 POLLOUT（可写）
// req: 请求信息
// ================================================================
void PollSocket::enqueue(int fd, short events, RequestInfo req) {
    pollfd pfd = {};             // 初始化 pollfd 结构
    pfd.fd = fd;                 // 设置要监控的 fd
    pfd.events = events;         // 设置监控事件（读/写）
    pfd.revents = 0;             // 清空返回事件
    
    pollfds.push_back(pfd);                           // 加入 poll 数组
    fdToRequest[fd] = std::move(req);                // 建立映射
    submitted++;                                      // 计数加一
}

// ================================================================
// 发送请求：监控 socket 可写
// req: 发送请求（包含 fd 和数据）
// msgFlags: 发送标志
// ================================================================
bool PollSocket::send(const Request& req, int32_t msgFlags) {
    RequestInfo info = {};
    info.request = const_cast<Request*>(&req);       // const 转非 const
    info.flags = msgFlags;
    info.timeout = std::chrono::steady_clock::now() + std::chrono::seconds(30);
    
    enqueue(req.fd, POLLOUT, std::move(info));       // 监控可写事件
    return true;
}

// ================================================================
// 接收请求：监控 socket 可读
// req: 接收请求（包含 fd 和缓冲区）
// msgFlags: 接收标志
// ================================================================
bool PollSocket::recv(Request& req, int32_t msgFlags) {
    RequestInfo info = {};
    info.request = &req;
    info.flags = msgFlags;
    info.timeout = std::chrono::steady_clock::now() + std::chrono::seconds(30);
    
    enqueue(req.fd, POLLIN, std::move(info));        // 监控可读事件
    return true;
}

// ================================================================
// 带超时的发送请求
// timeout: 超时时间
// ================================================================
bool PollSocket::send_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags) {
    RequestInfo info = {};
    info.request = const_cast<Request*>(&req);
    info.flags = msgFlags;
    info.timeout = std::chrono::steady_clock::now() + timeout;  // 使用传入的超时
    
    enqueue(req.fd, POLLOUT, std::move(info));
    return true;
}

// ================================================================
// 带超时的接收请求
// timeout: 超时时间
// ================================================================
bool PollSocket::recv_to(Request& req, std::chrono::milliseconds timeout, int32_t msgFlags) {
    RequestInfo info = {};
    info.request = &req;
    info.flags = msgFlags;
    info.timeout = std::chrono::steady_clock::now() + timeout;
    
    enqueue(req.fd, POLLIN, std::move(info));
    return true;
}

// ================================================================
// 获取已完成的请求
// 返回最早完成的请求，nullptr 表示没有
// ================================================================
Request* PollSocket::complete() {
    if (readyFds >= static_cast<int32_t>(ready.size())) {
        return nullptr;                                 // 没有更多完成的请求
    }
    return ready[readyFds++];                          // 返回并移动索引
}

// ================================================================
// 提交请求并等待事件
// 调用 poll() 系统调用，等待事件发生
// ================================================================
int32_t PollSocket::submit() {
    if (pollfds.empty()) {
        return 0;                                      // 没有监控的 fd
    }

    int timeoutMs = 1000;                              // poll 超时 1 秒
    int n = ::poll(pollfds.data(), pollfds.size(), timeoutMs);
    if (n < 0) {
        return -1;                                      // poll 错误
    }

    ready.clear();                                     // 清空已就绪队列
    readyFds = 0;

    // 遍历所有 pollfd，检查哪些有事件发生
    for (size_t i = 0; i < pollfds.size(); ++i) {
        if (pollfds[i].revents) {                      // 有事件发生
            int fd = pollfds[i].fd;
            auto it = fdToRequest.find(fd);
            if (it != fdToRequest.end()) {
                ready.push_back(it->second.request);   // 加入完成队列
                fdToRequest.erase(it);                  // 移除映射
            }
            // 移除这个 pollfd（用最后一个替换）
            pollfds[i] = pollfds.back();
            pollfds.pop_back();
            --i;                                        // 重新检查当前位置
        }
    }

    return static_cast<int32_t>(ready.size());        // 返回完成数量
}

}  // namespace myblob::network
```

### poll 多路复用原理

```
┌─────────────────────────────────────────────────────────┐
│                    PollSocket 工作流程                    │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  1. enqueue(fd, POLLOUT)    → 将 fd 加入 poll 监控     │
│     ┌─────────────────┐                              │
│     │ pollfds = [...]  │  ┌────────────────────────┐  │
│     │  fd1: POLLIN     │  │ fd1 → RequestInfo1   │  │
│     │  fd2: POLLOUT    │  │ fd2 → RequestInfo2   │  │
│     └─────────────────┘  └────────────────────────┘  │
│                                                         │
│  2. submit()              → 调用 poll() 等待事件        │
│     ┌─────────────────┐                              │
│     │ poll() 返回 n   │  n > 0: 有 n 个 fd 就绪     │
│     │ pollfds[i].revents   │  n = 0: 超时，无事件   │
│     └─────────────────┘  n < 0: 错误                 │
│                                                         │
│  3. complete()            → 返回已就绪的请求            │
│     ┌─────────────────┐                              │
│     │ ready = [req1]  │  ← 用户从队列取出请求处理     │
│     └─────────────────┘                              │
│                                                         │
└─────────────────────────────────────────────────────────┘

优点：单线程同时监控多个 socket，避免阻塞
```

## 5. ConnectionManager 实现

> 📁 文件：`src/network/connection_manager.cpp`（**修改现有文件**）

### 代码详解

```cpp
#include "network/connection_manager.hpp"
#include <fcntl.h>                   // fcntl 设置非阻塞
#include <netinet/tcp.h>             // TCP 选项
#include <sys/socket.h>             // socket API
#include <unistd.h>                 // close
#include <cstring>                   // memset

namespace myblob::network {

ConnectionManager::ConnectionManager() {
    // 初始化 poll socket
    pollSocket_ = std::make_unique<PollSocket>();
}

ConnectionManager::~ConnectionManager() {
    std::lock_guard<std::mutex> lock(mutex_);
    // 关闭所有连接
    for (auto& [fd, conn] : connections_) {
        if (conn.fd >= 0) {
            ::close(conn.fd);
        }
    }
    connections_.clear();
}

// ================================================================
// 创建 socket 并应用设置
// ================================================================
int ConnectionManager::createSocket(const std::string& hostname, uint16_t port, const TCPSettings& settings) {
    int fd = ::socket(AF_INET, SOCK_STREAM, 0);    // 创建 TCP socket
    if (fd < 0) {
        return -1;
    }
    
    // 设置非阻塞模式
    if (settings.nonBlocking) {
        int flags = fcntl(fd, F_GETFL, 0);
        fcntl(fd, F_SETFL, flags | O_NONBLOCK);
    }
    
    // 设置 TCP_NODELAY（禁用 Nagle 算法）
    if (settings.noDelay) {
        int flag = 1;
        setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &flag, sizeof(flag));
    }
    
    // 设置 SO_KEEPALIVE
    if (settings.keepAlive) {
        int flag = 1;
        setsockopt(fd, SOL_SOCKET, SO_KEEPALIVE, &flag, sizeof(flag));
    }
    
    applyTCPSettings(fd, settings);
    
    return fd;
}

// ================================================================
// 应用额外的 TCP 设置
// ================================================================
void ConnectionManager::applyTCPSettings(int fd, const TCPSettings& settings) {
    if (settings.recvBuffer > 0) {
        setsockopt(fd, SOL_SOCKET, SO_RCVBUF, &settings.recvBuffer, sizeof(settings.recvBuffer));
    }
    if (settings.mss > 0) {
        setsockopt(fd, IPPROTO_TCP, TCP_MAXSEG, &settings.mss, sizeof(settings.mss));
    }
    if (settings.reuse) {
        int flag = 1;
        setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &flag, sizeof(flag));
    }
    if (settings.linger >= 0) {
        struct linger ling = {1, settings.linger};
        setsockopt(fd, SOL_SOCKET, SO_LINGER, &ling, sizeof(ling));
    }
}

// ================================================================
// 建立连接
// ================================================================
int32_t ConnectionManager::connect(const std::string& hostname, uint16_t port, bool tls, const TCPSettings& settings) {
    int fd = createSocket(hostname, port, settings);
    if (fd < 0) {
        return -1;
    }
    
    // 设置服务器地址
    struct sockaddr_in addr = {};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(port);
    inet_pton(AF_INET, hostname.c_str(), &addr.sin_addr);
    
    // 非阻塞 connect
    int ret = ::connect(fd, (struct sockaddr*)&addr, sizeof(addr));
    if (ret < 0 && errno != EINPROGRESS) {
        ::close(fd);
        return -1;
    }
    
    std::lock_guard<std::mutex> lock(mutex_);
    
    Connection conn = {};
    conn.fd = fd;
    conn.tls = tls;
    conn.lastUsed = std::chrono::steady_clock::now();
    
    connections_[fd] = std::move(conn);
    
    // LRU 淘汰：超过最大连接数时移除最老的连接
    if (connections_.size() > maxCachedConnections) {
        auto oldest = connections_.begin();
        for (auto it = connections_.begin(); it != connections_.end(); ++it) {
            if (it->second.lastUsed < oldest->second.lastUsed) {
                oldest = it;
            }
        }
        if (oldest != connections_.end()) {
            disconnect(oldest->first);
        }
    }
    
    return fd;
}

// ================================================================
// 断开连接
// ================================================================
void ConnectionManager::disconnect(int32_t fd) {
    std::lock_guard<std::mutex> lock(mutex_);
    auto it = connections_.find(fd);
    if (it != connections_.end()) {
        if (it->second.fd >= 0) {
            ::close(it->second.fd);
        }
        connections_.erase(it);
    }
}

// ================================================================
// 获取连接
// ================================================================
ConnectionManager::Connection* ConnectionManager::getConnection(int32_t fd) {
    std::lock_guard<std::mutex> lock(mutex_);
    auto it = connections_.find(fd);
    if (it != connections_.end()) {
        it->second.lastUsed = std::chrono::steady_clock::now();  // 更新访问时间
        return &it->second;
    }
    return nullptr;
}

}  // namespace myblob::network
```

## 6. Transaction 添加批量执行

> 📁 文件：`include/cloud/transaction.hpp`（**修改现有文件**）

### 代码详解

```cpp
#pragma once
#include "../network/message_result.hpp"     // 消息结果
#include "../network/original_message.hpp"  // 原始消息
#include "../network/connection_manager.hpp" // 连接管理器（新增）
#include "provider.hpp"                      // 云服务商
#include <memory>                            // unique_ptr
#include <string>                            // 字符串
#include <utility>                           // forward
#include <vector>                            // vector

namespace myblob::cloud {

class Transaction {
public:
    // 消息向量类型
    using message_vector_type = std::vector<std::unique_ptr<network::OriginalMessage>>;
    
    Transaction();
    explicit Transaction(Provider* provider);
    ~Transaction();
    
    void setProvider(Provider* provider);
    
    // ================================================================
    // 添加 GET 请求（非回调版本）
    // remotePath: 远程路径
    // range: 数据范围 {offset, length}
    // result: 接收缓冲区
    // capacity: 缓冲区大小
    // ================================================================
    bool getObjectRequest(
        const std::string& remotePath,
        std::pair<uint64_t, uint64_t> range = {0, 0},
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    // ================================================================
    // 添加 GET 请求（回调版本）
    // 使用模板支持任意 callable
    // ================================================================
    template <typename Callback>
    bool getObjectRequest(
        Callback&& callback,
        const std::string& remotePath,
        std::pair<uint64_t, uint64_t> range = {0, 0},
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    // 添加 PUT 请求（非回调版本）
    bool putObjectRequest(
        const std::string& remotePath,
        const char* data,
        uint64_t size,
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    // 添加 PUT 请求（回调版本）
    template <typename Callback>
    bool putObjectRequest(
        Callback&& callback,
        const std::string& remotePath,
        const char* data,
        uint64_t size,
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    // 添加 DELETE 请求（非回调版本）
    bool deleteObjectRequest(
        const std::string& remotePath,
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    // 添加 DELETE 请求（回调版本）
    template <typename Callback>
    bool deleteObjectRequest(
        Callback&& callback,
        const std::string& remotePath,
        uint8_t* result = nullptr,
        uint64_t capacity = 0
    );
    
    message_vector_type& getMessages() { return messages_; }
    const message_vector_type& getMessages() const { return messages_; }
    
    // ================================================================
    // 同步执行所有请求
    // 使用 poll 多路复用批量处理
    // ================================================================
    void execute();
    
    // ================================================================
    // 异步执行所有请求（线程池版本）
    // 在后台线程执行，完成后调用 callback
    // ================================================================
    void executeAsync();
    
    // ================================================================
    // 异步执行（通用回调版本）
    // ================================================================
    template <typename Callback>
    void executeAsync(Callback&& callback);

private:
    Provider* provider_{nullptr};                              // 云服务商
    message_vector_type messages_;                            // 请求列表
    std::unique_ptr<network::ConnectionManager> connMgr_;    // 连接管理器（新增）
};

// ================================================================
// 模板函数：GET 请求（回调版本）
// Callback&& 支持左值和右值引用
// std::forward 完美转发保持参数类型
// ================================================================
template <typename Callback>
bool Transaction::getObjectRequest(
    Callback&& callback,
    const std::string& remotePath,
    std::pair<uint64_t, uint64_t> range,
    uint8_t* result,
    uint64_t capacity
) {
    if (!provider_) return false;
    
    // 创建回调消息，自动管理生命周期
    auto msg = std::make_unique<network::OriginalCallbackMessage<Callback>>(
        std::forward<Callback>(callback),                     // 完美转发回调
        provider_->getRequest(remotePath, range),            // HTTP 请求
        *provider_,
        result,
        capacity
    );
    
    messages_.push_back(std::move(msg));
    return true;
}

// ================================================================
// 模板函数：PUT 请求（回调版本）
// ================================================================
template <typename Callback>
bool Transaction::putObjectRequest(
    Callback&& callback,
    const std::string& remotePath,
    const char* data,
    uint64_t size,
    uint8_t* result,
    uint64_t capacity
) {
    if (!provider_) return false;
    
    auto object = std::string_view(data, size);
    auto msg = std::make_unique<network::OriginalCallbackMessage<Callback>>(
        std::forward<Callback>(callback),
        provider_->putRequest(remotePath, object),
        *provider_,
        result,
        capacity
    );
    
    msg->setPutRequestData(reinterpret_cast<const uint8_t*>(data), size);
    messages_.push_back(std::move(msg));
    return true;
}

// ================================================================
// 模板函数：DELETE 请求（回调版本）
// ================================================================
template <typename Callback>
bool Transaction::deleteObjectRequest(
    Callback&& callback,
    const std::string& remotePath,
    uint8_t* result,
    uint64_t capacity
) {
    if (!provider_) return false;
    
    auto msg = std::make_unique<network::OriginalCallbackMessage<Callback>>(
        std::forward<Callback>(callback),
        provider_->deleteRequest(remotePath),
        *provider_,
        result,
        capacity
    );
    
    messages_.push_back(std::move(msg));
    return true;
}

// ================================================================
// 异步执行（通用回调版本）
// 在新线程执行，执行完成后调用 callback
// ================================================================
template <typename Callback>
void Transaction::executeAsync(Callback&& callback) {
    // 使用 lambda 捕获，避免引用悬空
    std::thread([this, cb = std::forward<Callback>(callback)]() {
        execute();              // 执行请求
        cb();                   // 调用回调
    }).detach();                // 分离线程，不阻塞
}
```

```cpp
// ================================================================
// 同步执行所有请求
// 使用 poll 多路复用批量处理
// ================================================================
void Transaction::execute() {
    // 初始化连接管理器
    if (!connMgr_) {
        connMgr_ = std::make_unique<network::ConnectionManager>();
    }
    
    auto& pollSocket = connMgr_->getPollSocket();
    
    // 1. 建立所有连接
    for (auto& msg : messages_) {
        if (msg->result.getState() == network::MessageState::Init) {
            int fd = connMgr_->connect(
                msg->provider.getAddress(),
                msg->provider.getPort(),
                msg->provider.isHTTPS(),
                network::ConnectionManager::TCPSettings{}
            );
            
            if (fd >= 0) {
                msg->result.state.store(network::MessageState::Sending);
                
                // 添加到 poll 监控（可写 = 连接完成）
                network::Socket::Request req = {};
                req.fd = fd;
                req.event = network::Socket::EventType::write;
                req.userData = msg.get();
                pollSocket.send(req);
            } else {
                msg->result.state.store(network::MessageState::Aborted);
            }
        }
    }
    
    // 2. 等待连接完成，然后发送数据
    bool allDone = false;
    while (!allDone) {
        int32_t completed = pollSocket.submit();
        
        // 处理已完成的请求
        while (auto* req = pollSocket.complete()) {
            auto* msg = static_cast<network::OriginalMessage*>(req->userData);
            
            if (msg->result.getState() == network::MessageState::Sending) {
                // 发送 HTTP 请求
                ssize_t sent = ::send(req->fd, 
                                      msg->message->data(), 
                                      msg->message->size(), 0);
                
                if (sent > 0) {
                    msg->result.state.store(network::MessageState::Receiving);
                    
                    // 改为监控可读
                    network::Socket::Request recvReq = {};
                    recvReq.fd = req->fd;
                    recvReq.event = network::Socket::EventType::read;
                    recvReq.userData = msg;
                    pollSocket.recv(recvReq);
                }
            } else if (msg->result.getState() == network::MessageState::Receiving) {
                // 接收响应
                uint8_t buffer[8192];
                ssize_t received = ::recv(req->fd, buffer, sizeof(buffer), 0);
                
                if (received > 0) {
                    auto* dataVector = new utils::DataVector<uint8_t>();
                    dataVector->resize(received);
                    memcpy(dataVector->data(), buffer, received);
                    msg->setResultVector(dataVector);
                    msg->result.state.store(network::MessageState::Finished);
                    
                    if (msg->requiresFinish()) {
                        msg->finish();  // 调用回调
                    }
                    
                    connMgr_->disconnect(req->fd);
                } else if (received == 0) {
                    msg->result.state.store(network::MessageState::Finished);
                    if (msg->requiresFinish()) {
                        msg->finish();
                    }
                    connMgr_->disconnect(req->fd);
                }
            }
        }
        
        // 检查是否全部完成
        allDone = true;
        for (auto& msg : messages_) {
            auto state = msg->result.getState();
            if (state != network::MessageState::Finished &&
                state != network::MessageState::Aborted &&
                state != network::MessageState::Cancelled) {
                allDone = false;
                break;
            }
        }
        
        if (completed == 0 && !allDone) {
            std::this_thread::sleep_for(std::chrono::milliseconds(1));
        }
    }
}

// ================================================================
// 异步执行（默认空回调）
// ================================================================
void Transaction::executeAsync() {
    executeAsync([]() {});
}
```

```cpp
#include "cloud/provider.hpp"
#include "cloud/transaction.hpp"
#include <iostream>

// ================================================================
// 打印结果的回调函数
// ================================================================
void printResult(network::MessageResult& result, const std::string& path) {
    std::cout << "=== " << path << " ===" << std::endl;
    std::cout << "状态: " << static_cast<int>(result.getState()) << std::endl;
    std::cout << "成功: " << (result.success() ? "是" : "否") << std::endl;
    if (result.success()) {
        std::cout << "大小: " << result.getSize() << std::endl;
    } else {
        std::cout << "错误码: " << result.getFailureCode() << std::endl;
    }
    std::cout << std::endl;
}

int main() {
    // ================================================================
    // 1. 创建 Provider
    // ================================================================
    auto provider = myblob::cloud::Provider::makeProvider("http://example.com/");
    if (!provider) {
        std::cerr << "创建 Provider 失败" << std::endl;
        return 1;
    }
    
    // ================================================================
    // 2. 创建 Transaction
    // ================================================================
    myblob::cloud::Transaction transaction(provider.get());
    
    // 分配缓冲区
    uint8_t buffer1[1024];
    uint8_t buffer2[1024];
    uint8_t buffer3[1024];
    
    // ================================================================
    // 3. 添加多个 GET 请求
    // ================================================================
    transaction.getObjectRequest("/file1.txt", {0, 0}, buffer1, sizeof(buffer1));
    transaction.getObjectRequest("/file2.txt", {0, 0}, buffer2, sizeof(buffer2));
    transaction.getObjectRequest("/file3.txt", {0, 0}, buffer3, sizeof(buffer3));
    
    std::cout << "添加了 " << transaction.getMessages().size() << " 个请求" << std::endl;
    std::cout << "开始批量执行..." << std::endl;
    
    // ================================================================
    // 4. 同步执行
    // 使用 poll 多路复用，同时处理多个请求
    // ================================================================
    transaction.execute();
    
    // ================================================================
    // 5. 打印结果
    // ================================================================
    std::cout << "\n=== 执行结果 ===" << std::endl;
    int i = 1;
    for (auto& msg : transaction.getMessages()) {
        printResult(msg->result, "/file" + std::to_string(i++) + ".txt");
    }
    
    return 0;
}
```
